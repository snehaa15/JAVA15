public static boolean isPrime(int n) {
    if (n < 2) return false;
    for (int i = 2; i <= Math.sqrt(n); i++) {
        if (n % i == 0) return false;
    }
    return true;
}

public static void findTwinPrimes(int limit) {
    for (int i = 2; i < limit - 2; i++) {
        if (isPrime(i) && isPrime(i + 2)) {
            System.out.println("(" + i + ", " + (i + 2) + ")");
        }
    }
}

// Example usage:
findTwinPrimes(100);
