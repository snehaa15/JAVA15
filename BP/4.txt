class BPr4{
public static void main(String[] args){
int a=20;
int b=20;
int sum=a+b;
int sub=b-a;
int product=a*b;
int div=a/b;
int rem=a%b;
System.out.println("Number 1 is :  "+a+"  Number 2 is : "+b);

}
}
