class Employee {
    private String name;
    private double salary;
    private int hireYear;

    // Constructor
    public Employee(String name, double salary, int hireYear) {
        this.name = name;
        this.salary = salary;
        this.hireYear = hireYear;
    }

    // Getters
    public String getName() {
        return name;
    }

    public double getSalary() {
        return salary;
    }

    public int getHireYear() {
        return hireYear;
    }

    // Method to calculate years of service
    public int calculateYearsOfService(int currentYear) {
        return currentYear - hireYear;
    }

    // Method to print employee details
    public void printEmployeeDetails(int currentYear) {
        System.out.println("Name: " + name);
        System.out.println("Salary: $" + salary);
        System.out.println("Hire Year: " + hireYear);
        System.out.println("Years of Service: " + calculateYearsOfService(currentYear));
    }
}

class F {
    public static void main(String[] args) {
        Employee employee = new Employee("John Doe", 50000.0, 2010);

        int currentYear = 2022;
        employee.printEmployeeDetails(currentYear);
    }
}

OUTPUT


Name: John Doe
Salary: $50000.0
Hire Year: 2010
Years of Service: 12
